import { Navigate } from "react-router-dom";

// props:
// - allowedRoles: مصفوفة الأدوار المسموحة ["admin", "lecturer"]
// - role: دور المستخدم الحالي
// - children: المحتوى الداخلي المحمي

export default function RoleProtectedRoute({ allowedRoles, role, children }) {
  if (!allowedRoles.includes(role)) {
    return (
      <div className="text-center p-10 text-red-600 font-bold">
        ❌ غير مسموح لك بالوصول إلى هذه الصفحة
      </div>
    );
  }

  return children;
}
